```
#include<bits/stdc++.h>
using namespace std;

int main() {
	// i = 1
	// i = 2

	// i = 4
	// i = 5
	// i = 6
	int i;
	for(i = 1; i <= 5; i = i + 1) {

	}
	cout << i << endl;
	return 0;
}
```
This For-Loops example iterates the value enclosed inside of this function and prints out the output till the value is true and when it gets false the loop function comes to an end.
